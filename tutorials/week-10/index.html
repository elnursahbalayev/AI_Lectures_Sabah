1. Introduction

Training from scratch teaches you how CNNs work. Transfer learning teaches you how to use them in production. Today you do both.

We will build three image classifiers for the same binary task (Cars vs. Bikes) and compare them directly — a custom CNN trained from random weights, a frozen ResNet-18 used as a fixed feature extractor, and a fully fine-tuned ResNet-18. The gap in accuracy and data efficiency will make the power of transfer learning immediately obvious.

Today's Goal

  • Download and prepare a Cars vs. Bikes image dataset using torchvision.datasets.
  • Apply correct ImageNet normalization and data augmentation.
  • Build a custom ConvNet from scratch in PyTorch and train it.
  • Load a pretrained ResNet-18 and replace its classification head.
  • Compare feature extraction vs. full fine-tuning.
  • Visualize learned first-layer filters.
  • Implement Grad-CAM to highlight which image regions drive the prediction.
Task

class LabelRemappedSubset(torch.utils.data.Dataset):
    def __init__(self, base_dataset, indices, label_remap, transform=None):
        self.base      = base_dataset
        self.indices   = indices
        self.remap     = label_remap
        self.transform = transform

    def __len__(self):
        return len(self.indices)

    def __getitem__(self, idx):
        img, lbl = self.base[self.indices[idx]]
        if self.transform:
            img = self.transform(img)
        return img, self.remap[lbl]

4. Step 2 — Build a Custom CNN from Scratch

STEP 2 OF 9 • Architecture Design
class CustomCNN(nn.Module):
    """
    A small CNN: 3 convolutional blocks followed by a classification head.
    Input: (B, 3, 64, 64)
    """
    def __init__(self, num_classes=2):
        super().__init__()

        self.features = nn.Sequential(
            # Block 1: 3 → 32 channels, 64×64 → 32×32
            nn.Conv2d(3, 32, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),

            # Block 2: 32 → 64 channels, 32×32 → 16×16
            nn.Conv2d(32, 64, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),

            # Block 3: 64 → 128 channels, 16×16 → 8×8
            nn.Conv2d(64, 128, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(128),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(2, 2),
        )

        self.classifier = nn.Sequential(
            nn.AdaptiveAvgPool2d((1, 1)),   # (B, 128, 1, 1)
            nn.Flatten(),                   # (B, 128)
            nn.Linear(128, 64),
            nn.ReLU(inplace=True),
            nn.Dropout(0.4),
            nn.Linear(64, num_classes)
        )

    def forward(self, x):
        x = self.features(x)
        return self.classifier(x)

5. Step 3 — Training and Evaluation Functions

STEP 3 OF 9 • Training Infrastructure
def train_one_epoch(model, loader, criterion, optimizer, device):
    model.train()
    total_loss, correct, total = 0.0, 0, 0

    for imgs, labels in loader:
        imgs, labels = imgs.to(device), labels.to(device)

        optimizer.zero_grad()
        outputs = model(imgs)
        loss    = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        total_loss += loss.item() * imgs.size(0)
        correct    += (outputs.argmax(1) == labels).sum().item()
        total      += imgs.size(0)

    return total_loss / total, correct / total


def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct, total = 0.0, 0, 0
    all_preds, all_labels = [], []

    with torch.no_grad():
        for imgs, labels in loader:
            imgs, labels = imgs.to(device), labels.to(device)
            outputs = model(imgs)
            loss    = criterion(outputs, labels)

            total_loss += loss.item() * imgs.size(0)
            correct    += (outputs.argmax(1) == labels).sum().item()
            total      += imgs.size(0)
            all_preds.extend(outputs.argmax(1).cpu().numpy())
            all_labels.extend(labels.cpu().numpy())

    return total_loss / total, correct / total, all_preds, all_labels


def train_model(model, train_loader, test_loader, epochs=20,
                lr=1e-3, weight_decay=1e-4, label='Model'):
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(filter(lambda p: p.requires_grad, model.parameters()),
                           lr=lr, weight_decay=weight_decay)
    scheduler = optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=epochs)

    history = {'train_loss': [], 'val_loss': [],
               'train_acc':  [], 'val_acc':  []}

    print(f"\n{'='*60}")
    print(f"  Training: {label}")
    print(f"{'='*60}")

    for epoch in range(1, epochs + 1):
        tr_loss, tr_acc = train_one_epoch(model, train_loader, criterion, optimizer, device)
        va_loss, va_acc, _, _ = evaluate(model, test_loader, criterion, device)
        scheduler.step()

        history['train_loss'].append(tr_loss)
        history['val_loss'].append(va_loss)
        history['train_acc'].append(tr_acc)
        history['val_acc'].append(va_acc)

        if epoch % 5 == 0 or epoch == 1:
            print(f"  Epoch {epoch:3d}/{epochs} | "
                  f"Train Loss: {tr_loss:.4f}  Acc: {tr_acc:.1%} | "
                  f"Val Loss:   {va_loss:.4f}  Acc: {va_acc:.1%}")

    return history

def build_resnet_feature_extractor(num_classes=2):
    """
    Load pretrained ResNet-18, freeze all layers, replace final FC.
    Only the new classification head will be trained.
    """
    model = models.resnet18(weights='IMAGENET1K_V1')

    # Freeze all backbone parameters
    for param in model.parameters():
        param.requires_grad = False

    # Replace final FC layer (the only part that trains)
    in_features = model.fc.in_features   # 512 for ResNet-18
    model.fc = nn.Sequential(
        nn.Linear(in_features, 128),
        nn.ReLU(inplace=True),
        nn.Dropout(0.3),
        nn.Linear(128, num_classes)
    )

    trainable = sum(p.numel() for p in model.parameters() if p.requires_grad)
    total     = sum(p.numel() for p in model.parameters())
    print(f"ResNet-18 Feature Extractor:")
    print(f"  Total params:     {total:,}")
    print(f"  Trainable params: {trainable:,}  ({trainable/total:.1%} of model)")
    return model

7.1 Full Fine-Tuning

def build_resnet_finetune(num_classes=2):
    """
    Load pretrained ResNet-18, unfreeze all layers, use differential LRs.
    """
    model = models.resnet18(weights='IMAGENET1K_V1')
    in_features = model.fc.in_features
    model.fc = nn.Sequential(
        nn.Linear(in_features, 128),
        nn.ReLU(inplace=True),
        nn.Dropout(0.3),
        nn.Linear(128, num_classes)
    )
    # All parameters trainable
    total = sum(p.numel() for p in model.parameters())
    print(f"ResNet-18 Fine-Tune: {total:,} total params (all trainable)")
    return model

resnet_ft      = build_resnet_finetune().to(device)

# Differential learning rates: backbone gets 10× lower lr than head
criterion = nn.CrossEntropyLoss()
optimizer_ft = optim.Adam([
    {'params': [p for n, p in resnet_ft.named_parameters() if 'fc' not in n],
     'lr': 1e-5},
    {'params': resnet_ft.fc.parameters(),
     'lr': 1e-3},
], weight_decay=1e-4)

9. Step 7 — Visualize First-Layer Filters

STEP 7 OF 9 • What Does the Network See?
def visualize_filters(model, title='Conv Layer 1 Filters'):
    """Display the first-layer convolutional filters."""
    # Get the first conv layer weights
    first_conv = None
    for module in model.modules():
        if isinstance(module, nn.Conv2d):
            first_conv = module
            break

    if first_conv is None:
        print("No Conv2d found")
        return

    filters = first_conv.weight.detach().cpu()
    n = min(filters.shape[0], 32)   # show up to 32 filters

    # Normalize each filter for display
    f_min, f_max = filters[:n].min(), filters[:n].max()
    filters_norm = (filters[:n] - f_min) / (f_max - f_min + 1e-8)

    cols = 8
    rows = (n + cols - 1) // cols
    fig, axes = plt.subplots(rows, cols, figsize=(cols * 1.5, rows * 1.5))

    for i, ax in enumerate(axes.flat):
        if i < n:
            # Each filter is (C_in, k, k) → permute to (k, k, C_in) for imshow
            filt = filters_norm[i].permute(1, 2, 0).numpy()
            if filt.shape[2] == 1:
                ax.imshow(filt[:, :, 0], cmap='gray')
            else:
                filt = filt.clip(0, 1)
                ax.imshow(filt)
        ax.axis('off')

    plt.suptitle(title, fontsize=12, fontweight='bold')
    plt.tight_layout()
    plt.savefig('filters.png', dpi=100, bbox_inches='tight')
    plt.show()

10. Step 8 — Grad-CAM Visualization

STEP 8 OF 9 • Explainability

Grad-CAM (Gradient-weighted Class Activation Mapping) computes the gradient of the class score with respect to the last convolutional feature map. Regions with large positive gradients are highlighted — they contributed most to the predicted class.

class GradCAM:
    """Simple Grad-CAM implementation using forward/backward hooks."""
    def __init__(self, model, target_layer):
        self.model  = model
        self.grads  = None
        self.acts   = None

        target_layer.register_forward_hook(
            lambda m, i, o: setattr(self, 'acts', o.detach()))
        target_layer.register_full_backward_hook(
            lambda m, gi, go: setattr(self, 'grads', go[0].detach()))

    def generate(self, img_tensor, class_idx=None):
        self.model.eval()
        output = self.model(img_tensor.unsqueeze(0))

        if class_idx is None:
            class_idx = output.argmax(dim=1).item()

        self.model.zero_grad()
        output[0, class_idx].backward()

        # Global average pool the gradients
        weights = self.grads.mean(dim=(2, 3), keepdim=True)   # (1, C, 1, 1)
        cam = (weights * self.acts).sum(dim=1, keepdim=True)   # (1, 1, H, W)
        cam = torch.relu(cam)

        # Normalize
        cam = cam.squeeze().cpu().numpy()
        cam = (cam - cam.min()) / (cam.max() - cam.min() + 1e-8)
        return cam, class_idx, output.softmax(1)[0, class_idx].item()

# Attach Grad-CAM to the last conv layer of ResNet
target_layer = resnet_ft.layer4[-1].conv2
grad_cam = GradCAM(resnet_ft, target_layer)

What good Grad-CAM results look like:
  • For Cars: activation should concentrate on the body, wheels, or windshield — not the background.
  • For Bikes: activation should highlight handlebars, seat, or wheels.
  • If the heatmap focuses on the background instead of the object, the model has learned a spurious correlation — a warning sign that the dataset has a selection bias.

Save the state_dict, not the whole model. torch.save(model, path) pickles the entire class object — this breaks if you rename the class or move the file. Saving only model.state_dict() is robust and portable.

  • Experiment A — Deeper Backbone: Replace ResNet-18 with ResNet-50 (25M params vs. 11M). Does accuracy improve? How does training time change? Use models.resnet50(weights='IMAGENET1K_V2').
  • Experiment B — Data Efficiency: Train the fine-tuned ResNet-18 on only 20%, 40%, 60%, 80%, and 100% of the training data. Plot accuracy vs. dataset size for all three model types. At what dataset size does transfer learning provide the largest advantage?
  • Experiment C — Gradual Unfreezing: Start with only the FC head unfrozen. After 5 epochs, unfreeze layer4. After another 5, unfreeze layer3, and so on. Compare convergence speed and final accuracy against training all layers at once from epoch 1.
  • Experiment D — Adversarial Examples: Use Fast Gradient Sign Method (FGSM) to generate adversarial images: x_adv = x + ε * sign(∇_x L). Does the fine-tuned model break on these? Does it look like the correct class to you?
